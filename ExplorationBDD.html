import pandas as pd

def convert_csv_to_json(input_csv_path, output_json_path, target_columns, date_columns=None):
    # Charger les donnes depuis le fichier CSV en utilisant une tabulation comme dlimiteur
    data_frame = pd.read_csv(input_csv_path, delimiter='\t', header=None, names=["RawData"])
    
    # Diviser la colonne "RawData" en plusieurs colonnes en utilisant la tabulation comme sparateur
    parsed_data = data_frame["RawData"].str.split('\t', expand=True)
    
    # Renommer les colonnes avec les noms spcifis
    parsed_data.columns = target_columns

    # Si des colonnes de date sont spcifies, convertir les valeurs en format de date spcifi
    if date_columns:
        for date_col in date_columns:
            parsed_data[date_col] = pd.to_datetime(parsed_data[date_col], format='%d/%m/%y').dt.strftime('%d/%m/%y')

    # Exporter les donnes au format JSON avec une indentation de 2 espaces
    parsed_data.to_json(output_json_path, orient='records', indent=2)

import redis
import time

def publish_messages(channel):
    r = redis.Redis()
    message = "SALUT"

    while True:
        r.publish(channel, message)
        print(f"Message publi : {message}")
        time.sleep(3)  # Attendre 3 secondes

channel_name = "mon_canal"
publish_messages(channel_name)

import redis

def subscribe_to_channel(channel):
    r = redis.Redis()
    pubsub = r.pubsub()
    pubsub.subscribe(channel)

    print(f"Abonn au canal : {channel}")
    for message in pubsub.listen():
        if message['type'] == 'message':
            print(f"Message reu : {message['data'].decode()}")

Introduction aux Index dans MongoDB

Dans MongoDB, les index jouent un rle crucial dans l'optimisation des performances des requtes. Un index dans MongoDB est une structure de donnes spciale qui amliore la vitesse des oprations de recherche au sein des collections. Lorsqu'un index est cr sur un champ, MongoDB utilise cet index pour accder rapidement aux donnes sans avoir parcourir l'ensemble de la collection. Cela est particulirement utile pour les grandes bases de donnes, o les recherches sans index peuvent devenir lentes et coteuses en ressources.
